144  std::vector<size_type> GetDiagonal(const QnAxes &axes) {
145  std::vector<size_type> diagonal;
146  size_type stride_sum = 0;
147  size_type nbins_diagonal = 1;
148  size_type iaxis = 0;
149  for (const auto &ref_axis : axes_) {
150  bool found = false;
151  for (const auto &axis : axes) {
152  if (ref_axis==axis) {
153  stride_sum += stride_[iaxis];
154  found = true;
155  }
156  }
157  if (!found) nbins_diagonal *= ref_axis.size();
158  ++iaxis;
159  }
160  for (size_type ibin = 0; ibin < axes[0].size(); ++ibin) {
161  for (size_type idiag = 0; idiag < nbins_diagonal; ++idiag) {
162  diagonal.push_back(ibin*stride_sum + idiag);
163  }
164  }
165  return diagonal;
166  }

330  template<typename Function>
331  DataContainer<T> Projection(const std::vector<std::string> axis_names,
332  Function &&lambda) const {
333  DataContainer<T> projection;
334  unsigned long linearindex = 0;
335  std::vector<bool> isprojected;
336  isprojected.resize(axes_.size());
337  for (const auto &name : axis_names) {
338  bool bproj = false;
339  size_type iaxis = 0;
340  for (const auto &originalaxis : axes_) {
341  if (originalaxis.Name()==name) {
342  bproj = true;
343  break;
344  }
345  iaxis++;
346  }
347  isprojected.at(iaxis) = bproj;
348  }
349  size_type iaxis = 0;
350  for (const auto proj : isprojected) {
351  if (proj) projection.AddAxis(axes_.at(iaxis));
352  ++iaxis;
353  }
354  if (axis_names.empty()) {
355  projection.At(0) = data_.at(0);
356  for (auto bin = data_.begin() + 1; bin < data_.end(); ++bin) {
357  projection.At(0) = lambda(projection.At(0), *bin);
358  }
359  } else {
360  std::vector<size_type> indices;
361  std::vector<size_type> projindices;
362  indices.reserve(dimension_);
363  projindices.resize(projection.dimension_);
364  // other bins
365  for (auto bin : data_) {
366  this->GetIndex(indices, linearindex);
367  size_type iprojbin = 0;
368  for (size_type i = 0; i < indices.size(); ++i) {
369  if (isprojected.at(i)) {
370  projindices.at(iprojbin) = indices.at(i);
371  ++iprojbin;
372  }
373  }
374  projection.At(projindices) = lambda(projection.At(projindices), bin);
375  ++linearindex;
376  }
377  }
378  return projection;
379  }

475  DataContainer<T> Select(const Axis &axis) const {
476  DataContainer<T> selected;
477  long axisposition = 0;
478  long tmpaxisposition = 0;
479  for (const auto &a : axes_) {
480  if (a.Name()==axis.Name()) {
481  selected.AddAxis(axis);
482  axisposition = tmpaxisposition;
483  } else {
484  selected.AddAxis(a);
485  }
486  tmpaxisposition++;
487  }
488  unsigned long index = 0;
489  std::vector<unsigned long> indices;
490  indices.reserve(dimension_);
491  for (const auto &bin : data_) {
492  GetIndex(indices, index);
493  auto binlow = axes_[axisposition].GetLowerBinEdge(indices[axisposition]);
494  auto binhigh = axes_[axisposition].GetUpperBinEdge(indices[axisposition]);
495  auto binmid = binlow + (binhigh - binlow)/2;
496  auto rebinnedindex = axis.FindBin(binmid);
497  if (rebinnedindex!=-1) {
498  indices[axisposition] = static_cast<unsigned long>(rebinnedindex);
499  selected.CallOnElement(indices, [&bin](T &element) { element = bin; });
500  }
501  ++index;
502  }
503  if (axis.size()==1) {
504  selected.axes_.erase(selected.axes_.begin() + axisposition);
505  selected.stride_.resize(selected.axes_.size() + 1);
506  selected.dimension_ = selected.axes_.size();
507  selected.CalculateStride();
508  }
509  return selected;
510  }

519  template<typename Function>
520  DataContainer<T> Rebin(const Axis &rebinaxis, Function &&lambda) const {
521  DataContainer<T> rebinned;
522  unsigned long axisposition = 0;
523  bool axisfound = false;
524  // Check if axis to be rebinned is found in the datacontainer.
525  for (auto axis = std::begin(axes_); axis < std::end(axes_); ++axis) {
526  if (axis->Name()==rebinaxis.Name()) {
527  rebinned.AddAxis(rebinaxis);
528  axisposition = static_cast<unsigned long>(std::distance(axes_.begin(), axis));
529  axisfound = true;
530  } else {
531  rebinned.AddAxis(*axis);
532  }
533  }
534  if (!axisfound) {
535  std::string errormsg = "Datacontainer does not have axis of name " + rebinaxis.Name();
536  throw std::logic_error(errormsg);
537  }
538  //Check if there is no overlap in the bin edges.
539  bool rebin_ok = true;
540  for (const auto &rebinedge : rebinaxis) {
541  bool found = false;
542  for (const auto &binedge : (Axis) axes_.at(axisposition)) {
543  float test = TMath::Abs(rebinedge - binedge);
544  if (test < 10e-4) {
545  found = rebin_ok;
546  break;
547  }
548  }
549  rebin_ok = rebin_ok && found;
550  }
551  if (!rebin_ok) {
552  std::string errormsg = "Rebinned axis has overlapping bins." + rebinaxis.Name();
553  throw std::logic_error(errormsg);
554  }
555  unsigned long ibin = 0;
556  std::vector<size_type> indices;
557  indices.reserve(dimension_);
558  for (const auto &bin : data_) {
559  GetIndex(indices, ibin);
560  auto binlow = axes_[axisposition].GetLowerBinEdge(indices[axisposition]);
561  auto binhigh = axes_[axisposition].GetUpperBinEdge(indices[axisposition]);
562  auto binmid = binlow + (binhigh - binlow)/2;
563  auto rebinnedindex = rebinaxis.FindBin(binmid);
564  indices[axisposition] = static_cast<unsigned long>(rebinnedindex);
565  if (rebinnedindex!=-1) rebinned.At(indices) = lambda(rebinned.At(indices), bin);
566  ++ibin;
567  }
568  return rebinned;
569  }

606  template<typename Function>
607  DataContainer<T> Apply(const DataContainer<T> &data, Function &&lambda) const {
608  DataContainer<T> result;
609  std::vector<size_type> indices;
610  unsigned long index = 0;
611  if (axes_.size() > data.axes_.size()) {
612  for (unsigned long iaxis = 0; iaxis < data.axes_.size() - 1; ++iaxis) {
613  if (axes_[iaxis].Name()!=data.axes_[iaxis].Name()) {
614  std::string errormsg = "Axes do not match.";
615  throw std::logic_error(errormsg);
616  }
617  }
618  result.AddAxes(axes_);
619  indices.reserve(dimension_);
620  for (const auto &bin_a : data_) {
621  GetIndex(indices, index);
622  result.data_[index] = lambda(bin_a, data.At(indices));
623  ++index;
624  }
625  } else {
626  for (unsigned long iaxis = axes_.size() - 1; iaxis > 0; --iaxis) {
627  if (axes_[iaxis].Name()!=data.axes_[iaxis].Name()) {
628  std::string errormsg = "Axes do not match.";
629  throw std::logic_error(errormsg);
630  }
631  }
632  result.AddAxes(data.axes_);
633  indices.reserve(data.dimension_);
634  for (const auto &bin_b : data.data_) {
635  data.GetIndex(indices, index);
636  result.data_[index] = lambda(At(indices), bin_b);
637  ++index;
638  }
639  }
640  return result;
641  }

883 template<typename T>
884 DataContainer<T> ExclusiveSum(const DataContainer<T> &input) {
885  DataContainer<T> Summed(input);
886  Summed.ClearData();
887  for (auto ibin = std::begin(input); ibin < std::end(input); ++ibin) {
