48  void Connect() {
49  for (auto &event : subevents_) {
50  event.ConnectInput(input_treefile_.get(), input_file_.get());
51  if (event.GetState() > furthest_state_) {
52  furthest_state_ = event.GetState();
53  }
54  }
55  if (furthest_state_==EseSubEvent::State::collect) {
56  output_file_ = std::make_shared<TFile>(output_file_name_.data(), "NEW");
57  }
58  if (furthest_state_==EseSubEvent::State::calib) {
59  output_treefile_ = std::make_shared<TFile>(output_treefile_name_.data(), "NEW");
60  output_treefile_->cd();
61  output_tree_ = new TTree("ESE", "ESE");
62  }
63  for (auto &event : subevents_) {
64  event.ConnectOutput(output_tree_, output_file_.get());
65  }
66  SetupEventMatching();
67  }

82  bool Process(const std::vector<unsigned long> &eventindices) {
83  for (auto &event : subevents_) {
84  event.Do(eventindices);

92  void Finalize() {

96  if (furthest_state_==EseSubEvent::State::calib) {
97  if (output_treefile_ && output_tree_) {
98  output_treefile_->cd();

104  }
