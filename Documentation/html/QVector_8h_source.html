67  inline float x(const unsigned int i) const {
68  if (bits_.test(i)) {
69  return q_[std::bitset<kMaxNHarmonics>(bits_ & std::bitset<kMaxNHarmonics>((1 << (i + 1)) - 1)).count() - 1].x;
70  } else {
71  throw std::out_of_range("harmonic not in range.");
72  }
73  }

77  return q_[std::bitset<kMaxNHarmonics>(bits_ & std::bitset<kMaxNHarmonics>((1 << (i + 1)) - 1)).count() - 1].y;

79  throw std::out_of_range("harmonic not in range.");
80  }
