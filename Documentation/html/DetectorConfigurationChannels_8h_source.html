246 inline Bool_t DetectorConfigurationChannels::ProcessCorrections(const double *variableContainer) {
247 
248  /* first we build the raw Q vector with the chosen calibration */
249  BuildRawQnVector();
250 
251  /* then we transfer the request to the input data correction steps */
252  for (Int_t ixCorrection = 0; ixCorrection < fInputDataCorrections.GetEntries(); ixCorrection++) {
253  if (fInputDataCorrections.At(ixCorrection)->ProcessCorrections(variableContainer))
254  continue;
255  else
256  return kFALSE;
257  }
258 
259  /* input corrections were applied so let's build the Q vector with the chosen calibration */
260  BuildQnVector();
261 
262  /* now let's propagate it to Q vector corrections */
263  for (Int_t ixCorrection = 0; ixCorrection < fQnVectorCorrections.GetEntries(); ixCorrection++) {
264  if (fQnVectorCorrections.At(ixCorrection)->ProcessCorrections(variableContainer))
265  continue;
266  else
267  return kFALSE;
268  }
269  /* all correction steps were applied */
270  return kTRUE;
271 }

309 inline void DetectorConfigurationChannels::ClearConfiguration() {
310  /* transfer the order to the Q vector corrections */
311  for (Int_t ixCorrection = 0; ixCorrection < fQnVectorCorrections.GetEntries(); ixCorrection++) {
312  fQnVectorCorrections.At(ixCorrection)->ClearCorrectionStep();
313  }
314  /* transfer the order to the data vector corrections */
315  for (Int_t ixCorrection = 0; ixCorrection < fInputDataCorrections.GetEntries(); ixCorrection++) {
316  fInputDataCorrections.At(ixCorrection)->ClearCorrectionStep();
317  }
318  /* clean the raw Q vector */
319  fRawQnVector.Reset();
320  /* clean the own Q vector */
321  fPlainQnVector.Reset();
322  fPlainQ2nVector.Reset();
323  fCorrectedQnVector.Reset();
324  fCorrectedQ2nVector.Reset();
325  /* and now clear the the input data bank */
326  fDataVectorBank->Clear("C");
327 }
