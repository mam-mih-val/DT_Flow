148  inline bool CheckCuts(int i) {
149  int icut = 1;
150  if (cuts_.empty()) return true;
151  ++*((cut_weight_).begin() + i);// = *((cut_weight_).begin() + i) + 1.0;
152  bool passed = true;
153  for (auto &cut : cuts_) {
154  bool ipass = cut->Check(i) && passed;
155  if (ipass) {
156  ++*cut_weight_.at(i + nchannels_*icut);
157  }
158  passed = ipass;
159  ++icut;
160  }
161  return passed;
162  }

185  void CreateCutReport(std::string detname, std::size_t nchannels = 1) {
186  if (!cuts_.empty()) {
187  nchannels_ = nchannels;
188  auto offset = nchannels*(cuts_.size() + 1);
189  cut_i_ = Variable(0, offset);
190  cut_channel_ = Variable(offset, offset);
191  cut_weight_ = Variable(2*offset, offset);
192  var_values_ = new double[3*offset];
193  cut_weight_.var_container = var_values_;
194  cut_i_.var_container = var_values_;
195  cut_channel_.var_container = var_values_;
196  for (std::size_t i = 0; i < nchannels; ++i) {
197  for (std::size_t j = 0; j < (cuts_.size() + 1); ++j) {
198  var_values_[i + nchannels*j] = j;
199  var_values_[offset + i + nchannels*j] = i;
200  var_values_[2*offset + i + nchannels*j] = 0;
201  }
202  }
203  if (nchannels==1) {
204  std::string name = detname + "Cut_Report";
205  std::string title = std::string(";cuts;entries");
206  auto nbins = cuts_.size() + 1;
207  float low = 0.;
208  float high = cuts_.size() + 1;
209  auto histo = new TH1F(name.data(), title.data(), nbins, low, high);
210  int icut = 2;
211  histo->GetXaxis()->SetBinLabel(1, "all");
212  for (auto &cut : cuts_) {
213  histo->GetXaxis()->SetBinLabel(icut, cut->Name().data());
214  ++icut;
215  }
216  std::array<Variable, 2> arr = {{cut_i_, cut_weight_}};
217  report_ = std::make_unique<QAHisto1DPtr>(arr, histo);
218  } else {
219  std::string name = detname + "Cut_Report";
220  std::string title = std::string(";cuts;channels");
221  auto x_nbins = cuts_.size() + 1;
222  auto y_nbins = nchannels_;
223  float low = 0.;
224  float x_high = cuts_.size() + 1;
225  float y_high = nchannels_;
226  auto histo = new TH2F(name.data(), title.data(), x_nbins, low, x_high, y_nbins, low, y_high);
227  histo->GetXaxis()->SetBinLabel(1, "all");
228  int icut = 2;
229  for (auto &cut : cuts_) {
230  histo->GetXaxis()->SetBinLabel(icut, cut->Name().data());
231  ++icut;
232  }
233  std::array<Variable, 3> arr = {{cut_i_, cut_channel_, cut_weight_}};
234  report_ = std::make_unique<QAHisto2DPtr>(arr, histo);
235  }
236  }
237  }
