43 template<typename VAR, typename... T>
44 class EventCut : public EventCutBase {
45  public:
46  EventCut(VAR (&arr)[sizeof...(T)], std::function<bool(T...)> lambda)
47  : lambda_(lambda) {
48  int i = 0;
49  for (auto &a : arr) {
50  variables_.at(i) = std::move(a);
51  ++i;
52  }
53  }

60  bool Check() override {
61  return CheckImpl(std::make_index_sequence<sizeof...(T)>{});
62  }
63 
64  std::string Name() override {
65  std::string name;
66  for (unsigned int i = 0; i < variables_.size(); ++i) {
67  name+=variables_.at(i)->GetBranchName();
68  if (i < variables_.size() && variables_.size() != 1) {
69  name+="&&";
70  }
71  }
72  return name;
73  }

83  template<std::size_t... I>
84  bool CheckImpl(std::index_sequence<I...>) {
85  return lambda_(*(variables_[I]->Get())...);
86  }

89  std::function<bool(T...)> lambda_;
