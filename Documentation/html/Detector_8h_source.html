117  CorrectionDetector *GenerateDetector(int globalid, int binid, EventClassVariablesSet *set) override {
118  if (!configuration_) {
119  throw (std::runtime_error("No Qn correction configuration found for " + name_));
120  }
121  std::string name;
122  if (datavector_->IsIntegrated()) {
123  name = name_;
124  } else {
125  name = name_ + std::to_string(binid);
126  }
127  auto detector = new CorrectionDetector(name.data(), globalid);
128  auto configuration = CreateDetectorConfiguration(name, set);
129  configuration_(configuration);
130  normalization_ = configuration->GetQVectorNormalizationMethod();
131  detector->AddDetectorConfiguration(configuration);
132  return detector;

186  void FillData() override {
187  long i = 0;
188  if (!int_cuts_->CheckCuts(0)) return;
189  for (auto &histo : histograms_) {
190  histo->Fill();
191  }
192  for (const auto &phi : phi_) {
193  if (!cuts_->CheckCuts(i)) {
194  ++i;
195  continue;
196  }
197  if (vars_.empty()) {
198  datavector_->CallOnElement(0, [&](std::vector<DataVector> &vector) {
199  vector.emplace_back(phi, *(weight_.begin() + i));
200  });
201  } else {
202  long icoord = 0;
203  for (const auto &var : vars_) {
204  coordinates_.at(icoord) = *(var.begin() + i);
205  ++icoord;
206  }
207  try {
208  datavector_->CallOnElement(coordinates_,
209  [&](std::vector<DataVector> &vector) {
210  vector.emplace_back(phi, *(weight_.begin() + i));
211  });
212  }
213  catch (std::exception &) {}
214  }
215  ++i;
216  }
217  }
