149 inline void DetectorConfigurationTracks::BuildQnVector() {
150  fTempQnVector.Reset();
151  fTempQ2nVector.Reset();
152 
153  for (Int_t ixData = 0; ixData < fDataVectorBank->GetEntriesFast(); ixData++) {
154  CorrectionDataVector *dataVector = static_cast<CorrectionDataVector *>(fDataVectorBank->At(ixData));
155  fTempQnVector.Add(dataVector->Phi(), dataVector->Weight());
156  fTempQ2nVector.Add(dataVector->Phi(), dataVector->Weight());
157  }
158  /* check the quality of the Qn vector */
159  fTempQnVector.CheckQuality();
160  fTempQ2nVector.CheckQuality();
161  fTempQnVector.Normalize(fQnNormalizationMethod);
162  fTempQ2nVector.Normalize(fQnNormalizationMethod);
163  fPlainQnVector.Set(&fTempQnVector, kFALSE);
164  fPlainQ2nVector.Set(&fTempQ2nVector, kFALSE);
165  fCorrectedQnVector.Set(&fTempQnVector, kFALSE);
166  fCorrectedQ2nVector.Set(&fTempQ2nVector, kFALSE);
167 }

174 inline Bool_t DetectorConfigurationTracks::ProcessCorrections(const double *variableContainer) {
175  /* first we build the Q vector with the chosen calibration */
176  BuildQnVector();
177 
178  /* then we transfer the request to the Q vector correction steps */
179  /* the loop is broken when a correction step has not been applied */
180  for (Int_t ixCorrection = 0; ixCorrection < fQnVectorCorrections.GetEntries(); ixCorrection++) {
181  if (fQnVectorCorrections.At(ixCorrection)->ProcessCorrections(variableContainer))
182  continue;
183  else
184  return kFALSE;
185  }
186  /* all correction steps were applied */
187  return kTRUE;
188 }
