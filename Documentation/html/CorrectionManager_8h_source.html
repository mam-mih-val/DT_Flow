100  template<std::size_t N>
101  void AddDetector(std::string name,
102  Qn::DetectorType type,
103  const std::string &phi_name,
104  const std::string &weight_name,
105  const std::vector<Qn::Axis> &axes,
106  int const(&harmo)[N]) {
107  Variable phi = var_manager_->FindVariable(phi_name);
108  Variable weight = var_manager_->FindVariable(weight_name);
109  std::vector<Variable> vars;
110  vars.reserve(axes.size());
111  for (const auto &axis : axes) {
112  vars.push_back(var_manager_->FindVariable(axis.Name()));
113  }
114  std::unique_ptr<Detector<N>> det(new Detector<N>(name, type, axes, phi, weight, vars, harmo));
115  if (type==DetectorType::CHANNEL) {
116  detectors_channel_.emplace(name, std::move(det));
117  }
118  if (type==DetectorType::TRACK) {
119  detectors_track_.emplace(name, std::move(det));
120  }
121  }

132  template<std::size_t N, typename FUNCTION>
133  void AddCut(const std::string &name, const char *const (&names)[N], FUNCTION lambda) {
134  Variable arr[N];
135  int i = 0;
136  for (auto &n : names) {
137  arr[i] = var_manager_->FindVariable(n);
138  ++i;
139  }
140  auto cut = MakeUniqueNDimCut(arr, lambda);
141  if (detectors_track_.find(name)!=detectors_track_.end()) {
142  detectors_track_.at(name)->AddCut(std::move(cut));
143  } else if (detectors_channel_.find(name)!=detectors_channel_.end()) {
144  detectors_channel_.at(name)->AddCut(std::move(cut));
145  } else {
146  std::cout << "Detector" + name + "not found. Cut not Added." << std::endl;
147  }
148  }
