78 inline void CorrectionQnVectorBuild::Add(Double_t phi, Double_t weight) {
79 
80  if (weight < fMinimumSignificantValue) return;
81  for (Int_t h = 1; h < fHighestHarmonic + 1; h++) {
82  if ((fHarmonicMask & harmonicNumberMask[h])==harmonicNumberMask[h]) {
83  fQnX[h] += (weight*TMath::Cos(h*fHarmonicMultiplier*phi));
84  fQnY[h] += (weight*TMath::Sin(h*fHarmonicMultiplier*phi));
85  }
